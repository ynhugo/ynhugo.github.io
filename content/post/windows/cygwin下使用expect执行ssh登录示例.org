* 示例

** 示例1

#+begin_src bash
  #!/usr/bin/expect

  set timeout 5

  spawn ssh thc_2

  expect {
	  "yes/no" {
		  send "yes\r"
		  exp_continue
	  }
  }

  while {1} {
			expect {
				"Press" {
					send "\n"
					exp_continue
				}
				"y/N" {
					send "\n"
					exp_continue
				}
				"root" {
					break;
				}
			}		  
		}

		interact

#+end_src

*** 使用示例

#+begin_src bash
  expect expect-ssh.sh thc_2
#+end_src

** 示例2

#+begin_src bash
  #!/usr/bin/expect

  # 检查参数数量
  if {$argc < 1} {
		 puts "Usage: $argv0 <target_host>"
		 exit 1
	 }

	 set target_host [lindex $argv 0]

	 spawn ssh "$target_host"

	 expect {
		 "yes/no" {
			 send "yes\r"
			 exp_continue
		 }
	 }

	 while {1} {
			   expect {
				   "Press" {
					   send "\n"
					   exp_continue
				   }
				   "y/N" {
					   send "\n"
					   exp_continue
				   }
				   "root" {
					   break;
				   }
			   }		  
		   }

		   interact
#+end_src
