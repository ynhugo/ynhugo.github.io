* 麒麟系统安装Qt

#+begin_src bash
# cygwin
# ssh-keygen -R 192.168.3.104
sshpass -p 'Mind@123' ssh kylin@192.168.3.104

# kylin
sudo -v 
while true; do sudo -v; sleep 240; done &

# kylinV10-SP1安装Qt
# 1. 安装前的准备
sudo apt update -y;

# 1.1 判断 gcc  是否安装
gcc -v
# 没有安装的话就安装 gcc
sudo apt install -y gcc

# 1.2 判断 g++ 是否安装
g++ -v
# 没有安装的话就安装 g++
sudo apt install -y g++

# 1.3 判断 clang 是否安装，clang++通常包含在clang包里
clang -v; clang++ -v
# 没有安装的话就安装 clang
sudo apt install -y clang

# 1.4 判断 make 是否安装
make -v
# 没有安装的话就安装 make, make与make-guile互斥，从标准方案来看，建议安装make，不用管make-guile。因为build-essential是依赖make的
sudo apt install -y make

# 1.5 判断cmake是否安装
cmake -v
# 没有安装的话就安装cmake
sudo apt install -y cmake

# 2. 安装 QT5

# 2.1 安装 Qt5 的组件
sudo apt-get install -y build-essential

# 2.2 安装 Qt 的开发工具
sudo apt-get install -y qtbase5-dev qtchooser qt5-qmake qtbase5-dev-tools

# 2.3 安装 qtcreator
sudo apt-get install -y qtcreator

# 2.4 安装 qt
sudo apt-get install qt5*

sudo apt install -y libqt5charts5 libqt5charts5-dev
#+end_src

** 一键安装命令

#+begin_src bash
# cygwin
# ssh-keygen -R 192.168.3.104
# 麒麟系统需要在桌面环境下的网络相关设置中禁用防火墙
sshpass -p 'Mind@123' ssh kylin@192.168.3.104

# kylin
# 根据已知密码配置密码缓冲期
export SUDO_ASKPASS=$(mktemp)
cat > ${SUDO_ASKPASS} <<'EOF'
#!/bin/bash
# 输出密码
echo "Mind@123"
EOF
chmod 700 ${SUDO_ASKPASS}
sudo -A whoami
rm -f ${SUDO_ASKPASS}
while true; do sudo -v; sleep 240; done &

# 安装Qt环境

# 1、第1步：
sudo -v
# 2、第2步：
sudo -v && while true; do sudo -v; sleep 240; done &
# 3、第3步：
# 安装前的准备
sudo apt install -y \
     gcc \
     g++ \
     clang \
     make \
     cmake
# 安装Qt5
sudo apt-get install -y \
     build-essential \
     qtbase5-dev \
     qtchooser \
     qt5-qmake \
     qtbase5-dev-tools \
     qtcreator \
     qt5*
# 运行示例时报错修复
sudo apt install -y \
     qt5-default \
     libqt5serialport5-dev \
     qtdeclarative5-dev \
     qtquickcontrols2-5-dev

sudo apt install -y \
     aptitude \
     libgl1-mesa-glx \
     libgl1-mesa-dri \
     libglu1-mesa-dev \
     libgl1-mesa-dev \
     mesa-utils

sudo apt install -y \
     libqt5charts5 \
     libqt5charts5-dev

sudo ln -s /usr/lib/x86_64-linux-gnu/libGL.so.1 /usr/lib/libGL.so
#+end_src
