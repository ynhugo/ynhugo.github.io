** 华为昇腾开发套件Ubuntu22.04环境配置

#+begin_src bash
  sudo groupadd HwHiAiUser
  sudo useradd -g HwHiAiUser -d /home/HwHiAiUser -m HwHiAiUser -s /bin/bash
  sudo passwd HwHiAiUser
  usermod -a -G sudo HwHiAiUser
  sudo userdel -rf hwhiaiuser  
  # su HwHiAiUser
  reboot
  su - root
  echo 'HwHiAiUser    ALL=(ALL)       ALL' >> /etc/sudoers
  exit
  sudo apt update
  sudo apt install -y net-tools git emacs openssh-server vim fcitx5 fcitx5-*
  # 确保语言安装了中文简体并且中设置默认输入法为fcitx，然后重启
  reboot
#+end_src
