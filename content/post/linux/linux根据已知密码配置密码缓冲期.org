* 根据已知密码配置密码缓冲期

比如密码是：password@123

#+begin_src bash
# 根据已知密码配置密码缓冲期
export SUDO_ASKPASS=$(mktemp)
cat > ${SUDO_ASKPASS} <<'EOF'
#!/bin/bash
# 输出密码
echo "password@123"
EOF
chmod 700 ${SUDO_ASKPASS}
sudo -A whoami
rm -f ${SUDO_ASKPASS}
while true; do sudo -v; sleep 240; done &
#+end_src
