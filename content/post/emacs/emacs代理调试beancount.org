* emacs代理调试beancount

#+begin_src bash
python --version
<<'CONTENT'
Python 3.10.10
CONTENT

pip --version
<<'CONTENT'
pip 25.3 from D:\ProgramFiles\miniconda3\Lib\site-packages\pip (python 3.13)
CONTENT

pip install -i https://pypi.tuna.tsinghua.edu.cn/simple beancount fava beanprice

which bean-example
<<'CONTENT'
/cygdrive/d/ProgramFiles/miniconda3/Scripts/bean-example
CONTENT

temp="$HOME/temp"
mkdir -p $temp; cd $temp

cat - >> .emacs << 'EOF'
(add-to-list 'load-path "/path/to/beancount/src/elisp")
(require 'beancount)
(add-to-list 'auto-mode-alist '("\\.beancount\\'" . beancount-mode))

(load-theme 'misterioso t)
(if (>= emacs-major-version 29)
        (global-display-line-numbers-mode)
  (global-linum-mode t))
EOF

http_proxy="127.0.0.1:10808" HOME=$temp emacs
# 或
echo 'http_proxy="127.0.0.1:10808" HOME=$temp exec emacs' > bemacs
# 或
echo "HOME=$temp exec emacs" > bemacs
# 前边的exec会导致下一行的命令不会执行，因为不需要执行其他代码，如果需要则只去掉exec即可，如：echo "HOME=$temp emacs" > bemacs
# 然后
chmod +x bemacs
./bemacs


M-x，package-refresh-contents

M-x，package-install，beancount

# 重启emacs
C-x，C-c
emacs
# 或
http_proxy="127.0.0.1:10808" emacs

# 生成示例文件
bean-example > example.beancount
# web访问示例文件
fava example.beancount
#+end_src
