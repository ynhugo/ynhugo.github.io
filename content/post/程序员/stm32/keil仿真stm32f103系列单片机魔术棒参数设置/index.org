* 添加仿真串口配置文件

文件格式：.ini

文件内容

#+begin_src bash
  MODE COM1 115200,0,8,1
  ASSIGN COM1 <S1IN> S1OUT
#+end_src

* 重写输入/输出函数

重写输入/输出函数是在main.c文件中完成的

重写输入/输出函数之前需要先在main.c文件包含一下"stdio.h"头文件

#+begin_src c
  #include "stdio.h"
#+end_src

** 在main.c重写printf函数

#+begin_src c
  int fputc(int c, FILE * f)
  {		
	  int ch = 0;
	  ch=c;
	  HAL_UART_Transmit(&huart1,(uint8_t*)&ch,1,1000);//发送串口
	  return c;
  }
#+end_src

** 在main.c重写scanf函数

#+begin_src c
  int fgetc(FILE * F)    
  {
	  int ch_r = 0;
	  HAL_UART_Receive (&huart1,(uint8_t*)&ch_r,1,0xffff);//接收
	  return ch_r;
  }
#+end_src
