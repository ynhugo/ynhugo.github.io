* shell环境基于时间创建文件

#+begin_src bash
  #!/usr/bin/bash

  var1=`date | tr '年' ' ' | tr '月' ' ' | tr '日' ' ' |  awk '{print $1}'`
  var2=`date | tr '年' ' ' | tr '月' ' ' | tr '日' ' ' |  awk '{print $2}'`
  var3=`date | tr '年' ' ' | tr '月' ' ' | tr '日' ' ' |  awk '{print $3}'`

  touch $var1$var2$var3.txt

  echo "$var1$var2$var3: yenao" > $var1$var2$var3.txt
  echo "1. " >> $var1$var2$var3.txt
#+end_src

* windows环境基于时间创建文件

#+begin_src bash
  set var1=%date:~0,4%
  set var2=%date:~5,2%
  set var3=%date:~8,2%

  if not exist %var1%%var2%%var3%.txt (
  type nul > %var1%%var2%%var3%.txt
  echo %var1%%var2%%var3%: yenao > %var1%%var2%%var3%.txt
  echo 1. >> %var1%%var2%%var3%.txt
  )
#+end_src
