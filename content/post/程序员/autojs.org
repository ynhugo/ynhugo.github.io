* 操作

** 循环操作

*** 循环打印数字

#+begin_src javascript 
for (var i = 0; i < 5; i++) {
	(function(index) {
		setTimeout(function() {
			toast(index)
		}, index * 1000);
	})(i)
}
#+end_src

*** 循环点击指定坐标

#+begin_src javascript
for (var i = 0; i < 5; i++) {
	press(513, 743, 200);
}
#+end_src
