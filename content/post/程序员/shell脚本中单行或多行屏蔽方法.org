* shell脚本中单行或多行屏蔽方法

单行注释符号：#
#+begin_src bash
# echo "abc"
#+end_src 

** 多行注释：

*** 方法一：

#+begin_src bash
((0)) && {屏蔽块}
#+end_src
如
#+begin_src bash
((0)) && {
  echo "bbb"
  echo "ccc"
}
#+end_src

*** 方法二：COMMENT也可使用其他词替代，如 a,b,c等
#+begin_src bash
<<COMMENT
屏蔽块
COMMENT
#+end_src
如
#+begin_src bash
<<COMMENT
echo "bbb"
echo "ccc"
COMMENT
#+end_src

也可采用:<< 'COMMENT'
#+begin_src bash
<<'COMMENT'
echo "bbb"
echo "ccc"
COMMENT
#+end_src

*** 方法三：空格 + 单引号

#+begin_src bash
'
echo "bbb"
echo "ccc"
'
#+end_src

*** 方法四：!也可以换成任意字符，如a,b,c等等

#+begin_src bash
<<!
屏蔽块
!
#+end_src
如
#+begin_src bash
<<!
echo "bbb"
echo "ccc"
!
#+end_src

*** 方法五：

#+begin_src bash
if false; then
  屏蔽块
fi
#+end_src
如
#+begin_src bash
if false; then
echo "bbb"
echo "ccc"
fi
#+end_src
