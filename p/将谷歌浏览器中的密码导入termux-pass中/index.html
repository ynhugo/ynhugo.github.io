  mkdir -p temp

var1=$(pwd)
for ((i=0; i < ${#names[@]}; i++)) ; do
  # echo "${usernames[${i}]}"
  cd "${names[${i}]}"
  if [ "${usernames[${i}]}" == "" ]; then
    usernames[${i}]="passwd${i}"
  fi
  # echo "${usernames[${i}]}"
  echo "${passwords[${i}]}" > "${usernames[${i}]}"
  echo "login: ${usernames[${i}]}" >> "${usernames[${i}]}"
  echo "url: ${urls[${i}]}" >> "${usernames[${i}]}"
  gpg --yes -r ynpass -e "${usernames[${i}]}"
  rm "${usernames[${i}]}"
  cd $var1
done
