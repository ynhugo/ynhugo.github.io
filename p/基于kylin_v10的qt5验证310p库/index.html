#include "acl/acl.h"
#include <iostream>

int main() {
    aclError ret = aclInit(nullptr);
    if (ret != ACL_ERROR_NONE) {
        std::cerr << "ACL init failed, error code: " << ret << std::endl;
        return -1;
    } else {
        std::cout << "ACL init success" << std::endl;
        aclFinalize();
    }

    return 0;
}
